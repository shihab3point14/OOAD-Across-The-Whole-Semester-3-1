\chapter{EQUIPMENT USED}
\label{ch:eqp_used}
\begin{itemize}
    \item \textbf{Hardware:} Personal Computer / Laptop.
    \item \textbf{Software/Tools:} 
    \begin{itemize}
        \item \textbf{PlantUML:} Used as the primary modeling language to write and render the sequence diagrams.
        \item \textbf{PlantUML Viewer:} (e.g., PlantUML Web Server, Visual Studio Code with PlantUML extension) to compile the code into visual diagrams.
        \item \textbf{LaTeX Editor:} (e.g., Overleaf, TeXstudio) for compiling and formatting the final lab report.
    \end{itemize}
\end{itemize}
